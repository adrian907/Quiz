7. What is the output of this program if input given is ‘abcqfghqbcd’?

    class Input_Output {
        public static void main(String args[]) throws IOException {	 
            char c;
            BufferedReader obj = new BufferedReader(new InputStreamReader(System.in));
            do {
                c = (char) obj.read();
	        System.out.print(c);
            } while(c != 'q');
        }
    }
